var API_PATH_SIGNUP = '/signup'

function post(url, data, cb) {
  var xhr = new XMLHttpRequest()
  xhr.onload = function() {
    cb(JSON.parse(this.responseText))
  }
  xhr.open('POST', url)
  xhr.setRequestHeader("Content-Type", "application/json")
  xhr.send(JSON.stringify(data))
}

    console.log(res)
  })
}
